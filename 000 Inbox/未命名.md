// 1. 定义您的标签页数据
const myTabs = [
    {
        title: "本周工作日",
        scriptName: "tasksCalendar", 
        contentParams: {
            pages: "dv.pages().file.tasks.where(t => !t.tags.includes('#exclude'))",
            view: "week",
            firstDayOfWeek: "1",
            options: "style11 filter noProcess"
        }
    },
    {
        title: "全周",
        scriptName: "tasksCalendar",
        contentParams: {
            pages: "dv.pages().file.tasks.where(t => !t.tags.includes('#exclude'))",
            view: "week",
            firstDayOfWeek: "1",
            options: "style9 filter noProcess"
        }
    },
    {
        title: "月度",
        scriptName: "tasksCalendar",
        contentParams: {
            pages: "dv.pages().file.tasks.where(t => !t.tags.includes('#exclude'))",
            view: "month",
            firstDayOfWeek: "1",
            options: "style9 filter noProcess"
        }
    }
];

// 2. 准备传递给脚本的参数
const params = {
    tabs: myTabs,
    rootId: 'my-main-tabs'
};

// 3. 调用脚本，并增加错误捕获
try {
    dv.paragraph("DEBUG: 开始调用 dv.view...");
    const generatedHtml = await dv.view("Scripts/list-to-tabs-framework", params);
    dv.paragraph("DEBUG: dv.view 调用成功.");
    dv.paragraph("DEBUG: generatedHtml 的类型是: " + typeof generatedHtml);
    dv.paragraph("DEBUG: generatedHtml 的内容是: " + generatedHtml);

    // 4. 在顶层代码块中，使用 dv.el() 渲染这个 HTML 字符串
    if (generatedHtml && typeof generatedHtml === 'string') {
        dv.paragraph("DEBUG: 准备调用 dv.el() 渲染...");
        dv.el('div', generatedHtml);
        dv.paragraph("DEBUG: dv.el() 渲染完成.");
    } else {
        dv.paragraph('<p style="color: orange;">警告: generatedHtml 为空或不是字符串，无法渲染。</p>');
    }
} catch (error) {
    // 如果这里捕获到错误，说明问题出在 dv.view 调用或脚本内部
    dv.paragraph('<p style="color: red;">发生致命错误！</p>');
    dv.paragraph('<p style="color: red;">错误名称: ' + error.name + '</p>');
    dv.paragraph('<p style="color: red;">错误信息: ' + error.message + '</p>');
    dv.paragraph('<p style="color: red;">错误堆栈: <pre>' + error.stack + '</pre></p>');
}
